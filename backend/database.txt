-- Database: preconsultationdb
CREATE DATABASE IF NOT EXISTS preconsultationdb;
USE preconsultationdb;

-- 1. DOCTORS
-- Holds clinician details for login and appointment creation
CREATE TABLE IF NOT EXISTS doctors (
    id INT AUTO_INCREMENT PRIMARY KEY,
    full_name VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL, -- Use a hash (e.g., bcrypt) in production
    created_account TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 2. USERS (Optional)
-- For patients who choose to create an account (not strictly required for access_code flow)
CREATE TABLE IF NOT EXISTS users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    full_name VARCHAR(100),
    email VARCHAR(100) UNIQUE,
    created_account TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 3. APPOINTMENTS
-- The core link. Doctor creates this -> generates access_code.
CREATE TABLE IF NOT EXISTS appointments (
    id INT AUTO_INCREMENT PRIMARY KEY,
    doctor_id INT NOT NULL,
    user_id INT NULL, -- Nullable: allows anonymous/guest patients via code
    appointment_date DATETIME NOT NULL,
    access_code VARCHAR(10) UNIQUE NOT NULL, -- The code the patient enters
    status ENUM('scheduled', 'completed', 'cancelled') DEFAULT 'scheduled',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (doctor_id) REFERENCES doctors(id) ON DELETE CASCADE,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE SET NULL
);

-- 4. SESSIONS
-- Created when a patient enters a valid access_code
CREATE TABLE IF NOT EXISTS sessions (
    id INT AUTO_INCREMENT PRIMARY KEY,
    appointment_id INT NOT NULL,
    started_session_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    ended_session TIMESTAMP NULL,
    summary_text TEXT, -- Quick cache of the final result
    FOREIGN KEY (appointment_id) REFERENCES appointments(id) ON DELETE CASCADE
);

-- 5. MESSAGES
-- Stores the conversation history between 'user' and 'ai'
CREATE TABLE IF NOT EXISTS messages (
    id INT AUTO_INCREMENT PRIMARY KEY,
    session_id INT NOT NULL,
    sender ENUM('user', 'ai') NOT NULL,
    content TEXT NOT NULL,
    sent_messages_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (session_id) REFERENCES sessions(id) ON DELETE CASCADE
);

-- 6. SUMMARIES
-- Structured output generated by AI for the Doctor to review
CREATE TABLE IF NOT EXISTS summaries (
    id INT AUTO_INCREMENT PRIMARY KEY,
    session_id INT UNIQUE NOT NULL, -- One summary per session
    summary_content TEXT NOT NULL, -- JSON or formatted text
    created_summary TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (session_id) REFERENCES sessions(id) ON DELETE CASCADE
);
